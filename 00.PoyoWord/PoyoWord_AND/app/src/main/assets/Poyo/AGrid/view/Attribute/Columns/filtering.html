filtering

Description

  • ● 컬럼별 필터링의 사용유무와 ui 형태를 json object 형식으로 설정합니다.
  • ● SBGridProperties.filtering = true 는 필수 사항이며, filtering 설정없이 'datatype' 으로 filtering형식을 설정할 수 있습니다.

Value

			
	 * 컬럼속성 : filtering : {usemode : value1, uitype : value2, displayui : value3}
	   - value1 (선택) : [boolean] 그리드에서 filtering 기능을 사용시 해당 컬럼에서 사용 유무
		  → true (default)		: filtering 사용
		  → false			: filtering 미사용   
	   - value2 (선택) : [string] filter의 ui형태
		  → 'checklist' (default)	: 값 filtering
		  → 'number'		: 값 크기 비교
	   - value3 (선택) : [boolean] filter 아이콘 표시
	   	  → true (default)		: filtering ui 사용
		  → false			: filtering ui 미사용
Example 
			
	var datagrid;
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_data.gridData';
		SBGridProperties.filtering = true;
		SBGridProperties.selectmode = 'free';
		SBGridProperties.columns = [		
			{caption : ['기본정보','이름']		ref : 'name',		width : '80px',	style : 'text-align:center',	type : 'output',		filtering : { usemode : false}},
			{caption : ['기본정보','나이'],		ref : 'age',		width : '80pxx',	style : 'text-align:center',	type : 'input' , 		typeinfo :{maxlength : 5},	 round : {type : 'floor', position : -2},	datatype : 'number'},
			{caption : ['기본정보','국적'],		ref : 'nation',	width : '100px',	style : 'text-align:center',	type : 'combo' , 		typeinfo : {ref:'comboData', label:'label', value:'value' , itemcount : 4, 'displayui':true} },
			{caption : ['기본정보','도시'],		ref : 'nation2',	width : '100px',	style : 'text-align:center',	type : 'inputcombo' , 	typeinfo : {ref:'combofilteringData', label:'label', value:'value' , itemcount : 4}, filtering : {usemode : true, uitype : 'checklist', displayui : false}, disabled : true},
			{caption : ['기본정보','결혼여부'],	ref : 'married',	width : '80px',	style : 'text-align:center',	type : 'checkbox', 		typeinfo : {checkedvalue : 'T', uncheckedvalue : 'F'} }
		];
		datagrid = _SBGrid.create(SBGridProperties);		
	};
