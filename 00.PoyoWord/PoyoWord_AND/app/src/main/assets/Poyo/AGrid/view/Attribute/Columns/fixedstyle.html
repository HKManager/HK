fixedstyle

Description

  • ● 컬럼 헤더의 스타일을 설정합니다.

Value

			
	 * 컬럼속성 : fixedstyle : value1
	   - value1 (필수) : [string] 컬럼 헤더 스타일
Example 
			
	var datagrid;
	var SBGridProperties = {};

	function createGrid(){
	
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_data.gridData';
		SBGridProperties.selectmode = 'free';
		SBGridProperties.columns = [
			{caption : ['이름'],		ref : 'name',		width : '80px',	fixedstyle : 'text-align:right',				type : 'output' },
			{caption : ['나이'],		ref : 'age',		width : '80pxx',	fixedstyle : 'text-align:left',				type : 'input' , 		typeinfo :{maxlength : 5},	 round : {type : 'floor', position : -2}},
			{caption : ['국적'],		ref : 'nation',	width : '100px',	fixedstyle : 'text-align:center',			type : 'combo' , 		typeinfo : {ref:'comboData', label:'label', value:'value' , itemcount : 4, 'displayui':true} },
			{caption : ['도시'],		ref : 'nation2',	width : '100px',	fixedstyle : 'background-color:yellow'',		type : 'inputcombo' , 	typeinfo : {ref:'combofilteringData', label:'label', value:'value' , itemcount : 4},	 disabled : true},
			{caption : ['결혼여부'],	ref : 'married',	width : '80px',	fixedstyle : 'text-align:center',			type : 'checkbox', 		typeinfo : {checkedvalue : 'T', uncheckedvalue : 'F'}  }
		];
		datagrid = _SBGrid.create(SBGridProperties);		
	};
