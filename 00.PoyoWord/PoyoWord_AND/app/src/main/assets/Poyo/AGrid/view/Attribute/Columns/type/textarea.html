textarea

Description

  • ● 컬럼 type 중 textarea type을 설정합니다.
  • ● 다중라인의 입력이 가능하며, tooltip을 이용하면 줄바꿈된 결과를 볼 수 있습니다.

Value

			
	 * typeinfo : {maxlength : value1, autonext : value2,  oneclickedit : value3, textareawidth : value4}
	   - value1 (선택) : [number] 입력 가능한 최대 글자수
	   - value2 (선택) : [boolean] maxlength만큼 입력이 되면 자동으로 다음 셀로 포커스 이동 여부
		  → true			: 포커스 이동
		  → false (default)	: 포커스 이동하지 않음
	   - value3 (선택) : [boolean] 마우스 한 번의 클릭으로 편집 모드 활성화 여부
		  → true			: 마우스 한 번 클릭으로 편집
		  → false (default)	: 마우스 더블 클릭으로 편집
	   - value4 (선택) : [number] 활성화된 편집모드의 너비 (단위 : px)
Example 
		
	var datagrid;
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_data.gridData';
		SBGridProperties.tooltip = true;
		SBGridProperties.columns = [
			{caption : ['기본정보','이름'],		ref : 'name',	width : '100px',	style : 'text-align:center',	type : 'output'},
			{caption : ['기본정보','나이'],		ref : 'age',	width : '50px',	style : 'text-align:center',	type : 'input' , 	typeinfo : {maxlength:3 } },
			{caption : ['부수정보','주소'],		ref : 'addr',	width : '250px',	style : 'text-align:center',	type : 'textarea',	typeinfo : {maxlength:15, autonext : true,  oneclickedit : true, textareawidth : 300} },
			{caption : ['부수정보','전화번호'],	ref : 'tel',		width : '130px',	style : 'text-align:center', 	type : 'output'},
			{caption : ['부수정보','이메일'],	ref : 'email',	width : '200px',	style : 'text-align:center',	type : 'output'},
			{caption : ['부수정보','취미'],		ref : 'hobby',	width : '100px',	style : 'text-align:center', 	type : 'output',	hidden : true },
			{caption : ['부수정보','전공'],		ref : 'major',	width : '100px',	style : 'text-align:center',	type : 'output'}
		];
		datagrid = _SBGrid.create(SBGridProperties);
	};
