personalize (attribute)

Description

  • ● 그리드의 너비, 컬럼 위치, 숨김 등 그리드의 개인화를 설정할 수 있습니다.
  • ● 주의사항 : ie의 경우는 server에 올려서 테스트 하셔야 합니다. chrome은 html 파일로도 테스트 및 확인이 가능합니다.

Value


	  * 속   성 : SBGridProperties.autosavepsnldata = value1;
	   - value1 : [boolean] 그리드 refresh로 개인화 정보를 자동 저장할지 여부
		  → true			: 자동 저장
		  → false	(default)	: 자동 저장하지 않음
		  
	  * 속   성 : SBGridProperties.psnldataid = value1;
	   - value1 : [string] 개인화를 저장할 ID

		function loClear() 
		function hideColumns(){
			if (_.size(datagrid.getSelectedCols()) < 1 ) {
				alert('그리드의 셀을 선택하셔야 합니다.');
				return;
			}

			// 그리드 컬럼의 뒤쪽부터 숨겨서 index 의 변동을 막는다.
			$.each(datagrid.getSelectedCols().reverse(), function(key,index){
				datagrid.setColHidden(index, true);
			});
		}

		function showColumns(){
			for (var i=0 ; i < datagrid.getCols(); i++){
				if (datagrid.getColHidden(i)){
					datagrid.setColHidden(i, false);
				}
			}
		}

		function loClear() {
			localStorage.clear();
			datagrid.rebuild();
		}
