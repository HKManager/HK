style

Description

  • ● 컬럼의 스타일을 설정합니다.

Value

			
	 * 컬럼속성 : style : value1
	   - value1 (필수) : [string] : 컬럼 스타일
Example 
			
	var datagrid;
	var SBGridProperties = {};

			{caption : ['기본정보','국적'],		ref : 'nation',	width : '100px',	style : 'text-align:center',			type : 'combo' , 		typeinfo : {ref:'comboData', label:'label', value:'value' , itemcount : 4, 'displayui':true} },
			{caption : ['기본정보','도시'],		ref : 'nation2',	width : '100px',	style : 'background-color:yellow'',		type : 'inputcombo' , 	typeinfo : {ref:'combofilteringData', label:'label', value:'value' , itemcount : 4},	 disabled : true},

		];
