findCell

Description

  • ● 입력받은 검색값을 가지고 있는 행과 열 인덱스 번호를 반환합니다.

Parameter

	
	 * 메소드 : datagrid.findCell(param1, param2, param3, param4, param5, param6);
	   - param1 (필수) : [string]  검색할 값
	   - param2 (필수) : [number]  검색 범위의 첫 번째 행 인덱스
	   - param3 (선택) : [number]  검색 범위의 첫 번째 열 인덱스
	   - param4 (선택) : [boolean]  대소문자 구분 여부
	   	  → true                 : 대소문자 구분
		  → false(default)  : 대소문자 구분 안함
	   - param5 (선택) : [boolean]  전체일치/부분일치 구분 여부
	   	  → true                 : 전체일치
		  → false(default)  : 부분일치
	   - param6 (선택) : [string]  label을 비교할 것인지 value를 비교할 것인지 설정 (label/value/all)
	   	  → lavel(default)   : label 비교
		  → value                : value 비교
		  → all                     : 전체 비교

	function fnFindCell(){
		var data = $('#idx').val();
		var nRow = parseInt($('#nRow').val());
		var nCol = parseInt($('#nCol').val());
		var label = $('#label').val();
		var casesense, fullmatch;

		if($('#nRow').val() == ''){
			nRow = -1;
		}
		if($('#nCol').val() == ''){
			nCol = -1;
		}

		if($('#casesense').val() === 'true'){
			casesense = true;
		}else{
			casesense = false;
		}

		if($('#fullmatch').val() === 'true'){
			fullmatch = true;
		}else{
			fullmatch = false;
		}
		
		var cellIndex = datagrid.findCell(data, nRow, nCol, casesense, fullmatch, label);
		var index = document.getElementById('index');
		$('#index').text('검색한 셀의 행 인덱스 : '+cellIndex[0]+'열 인덱스 : '+cellIndex[1]+' ');
	}
