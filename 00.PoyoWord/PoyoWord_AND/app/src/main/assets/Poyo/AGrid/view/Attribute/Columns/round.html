round

Description

  • ● 데이터의 반올림 처리를 json object 형식으로 설정합니다.

Value

			
	 * 컬럼속성 : round : {type : value1, position : value2}
	   - value1 (필수) : [string] 반올림 형식
		  → 'floor'	: 버림
		  → 'ceil'		: 올림
		  → 'round'	: 반올림
	   - value2 (필수) : [number] 반올림할 소수점 자리수(음수로 설정)
Example 
			
	var datagrid;
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'roundData';
		SBGridProperties.selectmode = 'free';
		SBGridProperties.columns = [
			{caption : ['원본 데이터'],			ref : 'name',	width : '120px',	style : 'text-align:center',	type : 'output'},
			{caption : ['소수점 2자리 버림'],		ref : 'name',	width : '120px',	style : 'text-align:center',	type : 'output',		round : {type : 'floor', position : -2}},
			{caption : ['소수점 3자리 올림'],		ref : 'age',	width : '120px',	style : 'text-align:center',	type : 'output' , 		typeinfo :{maxlength : 5}, round : {type : 'ceil', position : -3} },
			{caption : ['소수점 1자리 반올림'],		ref : 'addr',	width : '140px',	style : 'text-align:center',	type : 'output',		round : {type : 'round', position : -1} }
		];			
		datagrid = _SBGrid.create(SBGridProperties);		
	};
