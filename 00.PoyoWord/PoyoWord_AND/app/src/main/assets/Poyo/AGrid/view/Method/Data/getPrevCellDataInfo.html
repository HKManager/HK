    function  fnGetPrevCellDataInfo(){
        var cellData = datagrid.getPrevCellDataInfo();
        $(".rowIndex").text(cellData.row);
        $(".colIndex").text(cellData.col);
        $(".cellData").text(cellData.data);
    };
