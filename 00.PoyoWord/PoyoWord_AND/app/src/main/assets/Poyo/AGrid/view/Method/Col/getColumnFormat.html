getColumnFormat

Description

  • ● 지정한 열의 포멧(format) 속성을 반환합니다.

Parameter

	
	 * 메소드 : datagrid.getColumnFormat(param1);
	    - param1 (필수) : [number] 열 인덱스
Example 
			
    var datagrid;	
    var SBGridProperties = {};

    function createGrid(){			

        SBGridProperties.parentid = 'SBGridArea';
        SBGridProperties.id = 'datagrid';
        SBGridProperties.jsonref = 'ct_data.resources';
        SBGridProperties.selectmode = 'free';

        SBGridProperties.columns = [
            {caption : ['기본정보','이름'],			ref : 'name',		width : '80px',	style : 'text-align:center',	type : 'input', format : {type:'string', rule:'@ "사원"'}},
            {caption : ['기본정보','생년월일'],		ref : 'birth',		width : '90px',		style : 'text-align:center',	type : 'datepicker',typeinfo : {locale : 'ko' , dateformat :'yymmdd'}, format : {type:'date', rule:'yyyy-mm-dd', origin : 'yyyymmdd',locale:'ko' }  },
            {caption : ['부수정보','기록'],			ref : 'time',		width : '70px',	style : 'text-align:center',	type : 'output', format : {type:'date', rule:'HH:mm:ss', origin : 'HHmmss' } },
            {caption : ['부수정보','점수'],			ref : 'score',		width : '70px',	style : 'text-align:center',	type : 'output', format : {type:'number', rule:'#,###.##'} },
            {caption : ['부수정보','주소'],			ref : 'addr',		width : '250px',	style : 'text-align:center',	type : 'textarea', format : {type:'custom', callback : fnCustomFormat}  }
        ];
        datagrid = _SBGrid.create(SBGridProperties);
        datagrid.bind('click', 'evnClick');
    };
		
	
    function evnClick(){
        var nCol = datagrid.getCol();
        var nFormat = datagrid.getColumnFormat(nCol);
        var str = "";

        if (datagrid.getCol() < 0){
            alert('포멧정보를 확인할 열을 선택해주세요.');
        }else{
            $("#colIndex").text(nCol);

            for(key in nFormat){
                str += key + " : \"" + nFormat[key] + "\", ";
            }

            str = str.slice(0, -2);
            $("#isFormat").text("{ "+str+" }");
        }
    };
