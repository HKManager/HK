getSelectedCellsInfo

Description

Parameter

	var SBGridProperties = {};

    function fnMethod(){
        var data = datagrid.getSelectedCellsInfo();
        $('.count').text(data.nCount);
        $('.sum').text(data.nSum);
        $('.avg').text(data.nAvg);
    };
