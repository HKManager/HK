datamergefalseskip

Description

  • ● 병합에서 제외할 컬럼을 설정합니다.
  • ● 주의사항 : mergecells 속성이 'byrestriccol' 로 설정되어 있어야 합니다.

Value

	
	 * 속   성 : SBGridProperties.datamergefalseskip = value1;	 * default : false
	   - value1 : [boolean] 현재 컬럼을 병합할지 여부
		  → true (default)			: 현재 컬럼을 병합
		  → false	: 현재 컬럼을 병합하지 않음
Example 
			
	var datagrid;	
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'grid_data';				
		SBGridProperties.mergecells = 'byrestriccol';
		SBGridProperties.datamergefalseskip = true;
		
		SBGridProperties.columns = [
			{caption : ['순번'],			ref : 'col1',	width : '50px',	style : 'text-align:center',	type : 'output'},
			{caption : ['보고상태'],		ref : 'col2',	width : '80px',	style : 'text-align:center',	type : 'output',	merge:false},
			{caption : ['역학조사여부'],	ref : 'col3',	width : '80px',	style : 'text-align:center',	type : 'output'},
			{caption : ['확대차수'],		ref : 'col4',	width : '80px',	style : 'text-align:center',	type : 'output',	merge:false},
			{caption : ['진행상태'],		ref : 'col5',	width : '120px',	style : 'text-align:center',	type : 'output',	merge:false},
			{caption : ['보건소명'],		ref : 'col6',	width : '200px',	style : 'text-align:center',	type : 'output'},
			{caption : ['접촉자수'],		ref : 'col7',	width : '100px',	style : 'text-align:center',	type : 'output',	merge:false}
		];
		datagrid = _SBGrid.create(SBGridProperties);
	};
