getColWidth

Description

  • ● 입력한 인덱스번호에 해당하는 컬럼의 너비를 반환합니다.

Parameter

	
	 * 메소드 : datagrid.getColWidth();
	   - param1 : 없음
Example 
			
	var datagrid;	
	$(document).ready(function(){
		var SBGridProperties = {};
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_data.resources';	

		SBGridProperties.columns = [
			{caption : ['이름'],		ref : 'name',		width : '100px',		style : 'text-align:center',	type : 'output'},
			{caption : ['나이'],		ref : 'age',		width : '50px',			style : 'text-align:center',	type : 'output'},
			{caption : ['성별'],		ref : 'gender',		width : '80px',			style : 'text-align:center',	type : 'output'},
			{caption : ['주소'],	  	ref : 'addr',		width : '250px',		style : 'text-align:center',	type : 'output'},
			{caption : ['전화번호'],	ref : 'tel',		width : '130px',		style : 'text-align:center',	type : 'output'},
			{caption : ['이메일'],		ref : 'email',		width : '200px',		style : 'text-align:center',	type : 'output'},
			{caption : ['취미'],	  	ref : 'hobby',		width : '100px',		style : 'text-align:center',	type : 'output'},
			{caption : ['전공'],	  	ref : 'major',		width : '100px',		style : 'text-align:center',	type : 'output'}
		];
		datagrid = _SBGrid.create(SBGridProperties);
	});					
		

	function getColWidth(){
		alert(datagrid.getColWidth(parseInt($('#idxValue').val())));
	};
