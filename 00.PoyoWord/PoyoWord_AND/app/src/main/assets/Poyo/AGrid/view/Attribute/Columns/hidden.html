hidden

Description

  • ● 컬럼의 숨김여부를 설정합니다.

Value

			
	 * 컬럼속성 : hidden : value1
	   - value1 (필수) : [boolean] 컬럼의 숨김 여부
		  → true			: 컬럼 숨김
		  → false (default)	: 컬럼 숨기지 않음
Example 
			
	var datagrid;
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_data.gridData';
		SBGridProperties.selectmode = 'free';
		SBGridProperties.columns = [
			{caption : ['기본정보','이름'],		ref : 'name',		width : '80px',	style : 'text-align:center',	type : 'output'},
			{caption : ['기본정보','나이'],		ref : 'age',		width : '80pxx',	style : 'text-align:center',	type : 'input' , 		typeinfo :{maxlength : 5},	 round : {type : 'floor', position : -2}},
			{caption : ['부수정보','취미'],		ref : 'hobby',		width : '100px',	style : 'text-align:center', 	type : 'output',		hidden : true},
			{caption : ['부수정보','전공'],		ref : 'major',		width : '100px',	style : 'text-align:center',	type : 'output'}
		];
		datagrid = _SBGrid.create(SBGridProperties);		
	};
