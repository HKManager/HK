setRowData

Description

  • ● 선택한 행에 데이터를 입력하거나 변경합니다.

Parameter

	
	 * 메소드 : datagrid.setRowData(param1, param2, param3);
	   - param1 (필수) : [number] 행 인덱스
	   - param2 (필수) : [json string] 입력할 데이터의 값({ref1:"value1",ref2 : "value2", ...} 의 형식으로 작성)
	   - param3 (선택) : [boolean] 데이터 갱신 후 그리드의 refresh 여부
		  → true (default)	: refresh 실행 
		  → false			: refresh 미실행

	function fnSetRowData(){							
		var nRow = datagrid.getRow();
		var data={academy : $('.input1').val(), addr : $('.input4').val(),  name : $('.input2').val(),
				phone : $('.input3').val(), curriculum : $('.input5').val(), class : $('.input6').val(),
				maxstudent : $('.input7').val(), term : $('.input8').val(), startday : $('.input9').val(),
				totalminute : $('.input10').val(),pay : $('.input11').val(), teacher : $('.input12').val()};

		datagrid.setRowData(nRow, data);
	};
