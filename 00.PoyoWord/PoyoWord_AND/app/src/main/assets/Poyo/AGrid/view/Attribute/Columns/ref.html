ref

Description

  • ● json data와 연결된 컬럼의 key명을 설정합니다.
  • ● 필수 컬럼 속성입니다.

Value

			
	 * 컬럼속성 : ref : value1
	   - value1 (필수) : [string] json key명
Example 
			
	var datagrid;
	var SBGridProperties = {};
	
	function createGrid(){
		
		SBGridProperties.parentid = 'SBGridArea';
		SBGridProperties.id = 'datagrid';
		SBGridProperties.jsonref = 'ct_datda.gridData';
		SBGridProperties.selectmode = 'free';
		SBGridProperties.columns = [
			{caption : ['기본정보','이름'],		ref : 'name',		width : '80px',	style : 'text-align:center',	type : 'output',		filering : { usemode : false}},
			{caption : ['기본정보','나이'],		ref : 'age',		width : '80px',	style : 'text-align:center',	type : 'input' , 		typeinfo :{maxlength : 5},	 round : {type : 'floor', position : -2}},
			{caption : ['기본정보','국적'],		ref : 'nation',		width : '100px',	style : 'text-align:center',	type : 'combo' , 		typeinfo : {ref:'comboData', label:'name', value:'code' , itemcount : 4, 'displayui':true} },
			{caption : ['기본정보','국적2'],	ref : 'nation2',	width : '100px',	style : 'text-align:center',	type : 'inputcombo' , 	typeinfo : {ref:'comboData', label:'name', value:'code' , itemcount : 4},	 disabled : true},
			{caption : ['기본정보','결혼여부'],	ref : 'married',	width : '80px',	style : 'text-align:center',	type : 'checkbox', 		typeinfo : {checkedvalue : 'T', uncheckedvalue : 'F'}  }
		];
		datagrid = _SBGrid.create(SBGridProperties);		
	};
